\section*{Problem 1: Transformation of a Normal RV}
If $x \sim \mathcal{N}[0, \sigma^{2}]$, find the PDF of $y = x^{2}$.

\begin{proof}[Solution]
The PDF of $x$ is $f_x(x) = \frac{1}{\sqrt{2\pi\sigma^2}} e^{-x^2 / (2\sigma^2)}$.
For $y > 0$, the CDF of $y$ is:
\[
F_y(y) = P(x^2 \le y) = P(-\sqrt{y} \le x \le \sqrt{y}) = F_x(\sqrt{y}) - F_x(-\sqrt{y})
\]
Differentiating with respect to $y$ using the Chain Rule:
\[
f_y(y) = \frac{d}{dy} [F_x(\sqrt{y}) - F_x(-\sqrt{y})] = f_x(\sqrt{y}) \frac{1}{2\sqrt{y}} + f_x(-\sqrt{y}) \frac{1}{2\sqrt{y}}
\]
Since the normal PDF is symmetric about 0, $f_x(\sqrt{y}) = f_x(-\sqrt{y})$:
\[
f_y(y) = \frac{1}{\sqrt{y}} f_x(\sqrt{y}) = \frac{1}{\sqrt{y}} \frac{1}{\sqrt{2\pi\sigma^2}} e^{-(\sqrt{y})^2 / (2\sigma^2)} = \frac{1}{\sqrt{2\pi\sigma^2 y}} \exp\left\{-\frac{y}{2\sigma^2}\right\}, \quad y \ge 0
\]
\end{proof}

\section*{Problem 2: Probability of Independent Exponential RVs}
RVs $x$ and $y$ are independent and exponentially distributed with parameters $a$ and $b$, respectively. Calculate $P(\{x < y\})$.

\begin{proof}[Solution]
The joint PDF is $f_{xy}(x, y) = f_x(x)f_y(y) = a e^{-ax} \cdot b e^{-by}$ for $x, y > 0$.
\begin{align*}
P(x < y) &= \int_0^\infty \int_0^y a e^{-ax} b e^{-by} \, dx \, dy \\
&= \int_0^\infty b e^{-by} \left( \int_0^y a e^{-ax} \, dx \right) dy = \int_0^\infty b e^{-by} \left( 1 - e^{-ay} \right) dy \\
&= \int_0^\infty b e^{-by} \, dy - \int_0^\infty b e^{-(a+b)y} \, dy = 1 - \frac{b}{a+b} = \frac{a}{a+b}
\end{align*}
\end{proof}

\section*{Problem 3: Order Statistics}
RVs $x_1, x_2, x_3$ are independent and uniformly distributed on $[0, 1]$.
\begin{enumerate}
    \item[(a)] Define $y = \max(x_1, x_2, x_3)$. Find the PDF $f_y$ and calculate $E[y]$.
    \item[(b)] Define $z = \min(x_1, x_2, x_3)$. Find the PDF $f_z$ and $E[z]$.
\end{enumerate}

\begin{proof}[Solution]
For $x_i \sim U[0, 1]$, the CDF is $F_x(x) = x$ and the PDF is $f_x(x) = 1$ for $x \in [0, 1]$. \\
\textbf{(a)} $F_y(y) = P(x_1 \le y, x_2 \le y, x_3 \le y) = [F_x(y)]^3 = y^3$.
\[ f_y(y) = 3y^2, \quad 0 \le y \le 1; \quad E[y] = \int_0^1 y(3y^2) \, dy = \left[ \frac{3y^4}{4} \right]_0^1 = \frac{3}{4} \]
\textbf{(b)} $P(z > z) = P(x_1 > z, x_2 > z, x_3 > z) = [1 - F_x(z)]^3 = (1-z)^3$.
\[ F_z(z) = 1 - (1-z)^3 \implies f_z(z) = 3(1-z)^2, \quad 0 \le z \le 1; \quad E[z] = \frac{1}{4} \]
\end{proof}

\section*{Problem 4: Discrete RVs and Independence}
\begin{enumerate}
    \item[(a)] $x \in \{-2, -1, 0, 1, 2\}$ with $P=1/5$. $y = x^2 - 1$. Find statistics and independence.
    \item[(b)] $u \sim \mathcal{N}(\mu, \sigma^2)$, $v = u^2 - 1$. Find $E[v]$.
    \item[(c)] $z \in \{+1, -1\}$ with $P=1/2$, $w = z^2 - 1$. Are $z, w$ independent?
\end{enumerate}

\begin{proof}[Solution]
\textbf{(a)} $E[x] = 0$. $E[x^2] = \frac{1}{5}(4+1+0+1+4) = 2$, so $var(x) = 2$. \\
$y \in \{3, 0, -1, 0, 3\}$. $E[y] = 2-1 = 1$. $E[y^2] = \frac{1}{5}(3^2 \times 2 + 0^2 \times 2 + (-1)^2 \times 1) = 3.8$. $var(y) = 3.8 - 1^2 = 2.8$. \\
$x, y$ are \textbf{not independent} because $y$ is a function of $x$. For example, $P(x=0, y=3) = 0 \neq P(x=0)P(y=3)$. \\
\textbf{(b)} $E[v] = E[u^2] - 1 = (var(u) + E[u]^2) - 1 = \sigma^2 + \mu^2 - 1$. \\
\textbf{(c)} $w = z^2 - 1$. Since $z = \pm 1$, $z^2 = 1$ always, so $w = 0$ (constant). A constant is independent of any RV.
\end{proof}

\section*{Problem 5: Hypothesis Testing}
Measurements $x_k$ are i.i.d. $H_0: x_k \sim U[0, 1]$. $H_1: f_x(x) = 20x^{19}$ for $x \in [0, 1]$.
\begin{enumerate}
    \item[(a)] Binary quantization $b_k = 1$ if $x_k > T$. Find $T$ for $P(b_k=1|H_0) = 0.1$.
    \item[(b)] Global decision $H_1$ if $n = \sum b_k \ge 8$. Find $P_d$ and $P_f$.
\end{enumerate}

\begin{proof}[Solution]
\textbf{(a)} $P(x_k > T | H_0) = 1 - T = 0.1 \implies T = 0.9$. \\
$P(b_k=1 | H_1) = \int_{0.9}^1 20x^{19} \, dx = [x^{20}]_{0.9}^1 = 1 - 0.9^{20} \approx 0.887$. \\
\textbf{(b)} $P_f = P(n \ge 8 | H_0)$ where $n \sim Binom(10, 0.1)$.
\[ P_f = \binom{10}{8}(0.1)^8(0.9)^2 + \binom{10}{9}(0.1)^9(0.9) + (0.1)^{10} \approx 3.736 \times 10^{-7} \]
$P_d = P(n \ge 8 | H_1)$ where $n \sim Binom(10, 0.887) \approx 0.887$.
\end{proof}
